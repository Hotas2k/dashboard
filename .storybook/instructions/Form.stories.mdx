# Form

We are using [React Hook Form](https://react-hook-form.com/) when creating forms. When validating we're using [validator.js](https://www.npmjs.com/package/validator) and some homemade regex.

## Simple setup

```tsx
import { useForm } from 'react-hook-form';

import { TextField } from 'components/ui/forms';
import { useValidate } from 'hooks';

const Component = () => {
  const { isEmail } = useValidate();
  const {
    register,
    handleSubmit,
    formState: { errors },
    reset
  } = useForm();

  const onSubmit = handleSubmit((data) => {
    console.log(data);
    reset();
  });

  return (
    <form onSubmit={onSubmit}>
      <TextField
        name="email"
        type="email"
        label="Email"
        register={register}
        validation={{
          required: true,
          validate: isEmail
        }}
        error={errors.email}
      />
    </form>
  );
};
```
